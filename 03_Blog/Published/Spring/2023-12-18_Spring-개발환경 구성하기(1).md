## 9. 톰캣 서버 실행 코드 작성
```java
import java.io.File;

import org.apache.catalina.startup.Tomcat;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

public class WebApplicationServer {
    private static final Logger log = LoggerFactory.getLogger(WebApplicationServer.class);

    public static void main(String[] args) throws Exception {
        String webappDirLocation = "webapps/";  // 웹 애플리케이션의 위치
        Tomcat tomcat = new Tomcat();  // 톰캣 객체 생성
        tomcat.setPort(8080); // 포트 설정

        tomcat.addWebapp("/", new File(webappDirLocation).getAbsolutePath()); // 웹 애플리케이션 등록
        log.info("configuring app with basedir: {}", new File("./" + webappDirLocation).getAbsolutePath()); // 로그 출력

        tomcat.start(); // 톰캣 실행
        tomcat.getServer().await(); // 톰캣 서버 대기
    }
}
```
